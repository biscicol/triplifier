DROP TABLE IF EXISTS identifiers;
CREATE TABLE identifiers (
    id bigint NOT NULL UNIQUE AUTO_INCREMENT,       # the internal ID, represented in integer form
    ezid boolean DEFAULT false,                     # indicates whether the EZID has been made (ark://prefix/encode(id))
    doi text,                                       # the DOI
    webaddress text,                                # The webaddress target (_target per EZIDs)
    localid text,                                   # the local identifier
    what text,                                      # erc.what
    ts TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP    # erc.when
);
CREATE INDEX id_index ON identifiers (id);
